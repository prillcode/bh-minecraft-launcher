```ts
export interface Server {
  id: string;
  name: string;       // display name, e.g. "BlockHaven SMP"
  host: string;       // hostname or IP
  port: number;       // default 25565
  notes?: string;     // optional freeform notes
  createdAt: number;  // Date.now()
}

export interface ServerPingResult {
  online: boolean;
  latencyMs?: number;
  version?: string;       // e.g. "1.21.4"
  motd?: string;          // plain text, § codes stripped
  onlinePlayers?: number;
  maxPlayers?: number;
  error?: string;         // error message if online=false
}
```

<task type="auto">
  <name>Task 2: Create ServerManager class</name>
  <files>src/core/game/server.ts</files>
  <action>
Create a new file mirroring the InstanceManager pattern exactly. Use electron-store with `name: 'servers'`, `defaults: { servers: {} }`.

Store type: `Store<{ servers: Record<string, Server> }>`

Methods:
- `list(): Promise<Server[]>` — return `Object.values(all).sort((a, b) => a.name.localeCompare(b.name))`
- `get(id: string): Promise<Server>` — throw `new Error('Server not found: ' + id)` if missing
- `create(config: { name: string; host: string; port?: number; notes?: string }): Promise<Server>` — generate UUID with `randomUUID()`, default port to 25565, set `createdAt: Date.now()`, store and return
- `update(id: string, updates: Partial<Server>): Promise<Server>` — merge with existing, never allow id override (spread `...updates, id`)
- `delete(id: string): Promise<void>` — call `this.store.delete('servers.' + id as any)` (no filesystem cleanup needed, unlike InstanceManager)

Imports: `import Store from 'electron-store'`, `import { randomUUID } from 'crypto'`, `import type { Server } from './types'`, `import { logger } from '../utils/logger'`.

Log each mutation with `logger.info(...)` matching the style in instance.ts.
