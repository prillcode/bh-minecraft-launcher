  <files>src/renderer/components/instances/InstanceList.tsx, src/renderer/styles/globals.css</files>
  <action>
**InstanceList.tsx:**
1. Add state:
   ```ts
   const [servers, setServers] = useState<ServerInfo[]>([]);
   const [pingResults, setPingResults] = useState<Record<string, ServerPingResult>>({});
   ```
2. In the existing `useEffect`, also fetch servers:
   ```ts
   window.launcher.servers.list().then(setServers);
   ```
3. Add a `useMemo` for quick lookup:
   ```ts
   const serverMap = useMemo(
     () => Object.fromEntries(servers.map((s) => [s.id, s])),
     [servers]
   );
   ```
4. After servers load, ping each one. Add a second `useEffect` that fires when `servers` changes:
   ```ts
   useEffect(() => {
     if (servers.length === 0) return;
     Promise.all(
       servers.map((s) =>
         window.launcher.servers.ping(s.host, s.port).then((result) =>
           setPingResults((prev) => ({ ...prev, [s.id]: result }))
         )
       )
     );
   }, [servers]);
   ```
   (Fire-and-forget per server so they update the UI as each comes in.)

5. In the instance card JSX, replace the existing `inst.serverAutoConnect` block with:
   ```tsx
   {inst.serverId && serverMap[inst.serverId] && (
     <span className="instance-card__server">
       {serverMap[inst.serverId].name}
       {pingResults[inst.serverId] ? (
         pingResults[inst.serverId].online
           ? <span className="server-badge server-badge--online">
               {pingResults[inst.serverId].onlinePlayers}/{pingResults[inst.serverId].maxPlayers}
               {' · '}{pingResults[inst.serverId].latencyMs}ms
             </span>
           : <span className="server-badge server-badge--offline">Offline</span>
       ) : (
         <span className="server-badge server-badge--pinging">...</span>
       )}
     </span>
   )}
   ```
