**State:**
```ts
const [servers, setServers] = useState<ServerInfo[]>([]);
const [pingResults, setPingResults] = useState<Record<string, ServerPingResult>>({});
const [pinging, setPinging] = useState(false);
const [showAddModal, setShowAddModal] = useState(false);
const [editingServer, setEditingServer] = useState<ServerInfo | null>(null);
```

**On mount:** fetch servers then ping all.

**`pingAll()` function:**
```ts
const pingAll = async () => {
  if (servers.length === 0) return;
  setPinging(true);
  await Promise.all(
    servers.map((s) =>
      window.launcher.servers.ping(s.host, s.port).then((result) =>
        setPingResults((prev) => ({ ...prev, [s.id]: result }))
      )
    )
  );
  setPinging(false);
};
```

**`handleDelete(server: ServerInfo)`:** `confirm('Delete "' + server.name + '"?')`, call `window.launcher.servers.delete(server.id)`, remove from state.

**`handleAdd(server: ServerInfo)`:** prepend to servers, then ping just the new server.

**`handleUpdate(updated: ServerInfo)`:** replace matching entry in servers array.

**JSX structure:**
```tsx
<div className="servers">
  <div className="servers__header">
    <h2>Servers</h2>
    <div className="servers__actions">
      <button className="btn btn--ghost" onClick={pingAll} disabled={pinging}>
        {pinging ? 'Refreshing...' : 'Refresh'}
      </button>
      <button className="btn btn--primary" onClick={() => setShowAddModal(true)}>
        + Add Server
      </button>
    </div>
  </div>

  {servers.length === 0 ? (
    <div className="servers__empty">
      <p>No servers saved yet.</p>
      <button className="btn btn--primary" style={{ marginTop: '16px' }} onClick={() => setShowAddModal(true)}>
        Add your first server
      </button>
    </div>
  ) : (
    <div className="servers__list">
      {servers.map((server) => {
        const ping = pingResults[server.id];
        return (
          <div key={server.id} className="server-row">
            <div className="server-row__indicator">
              {ping === undefined
                ? <span className="server-row__dot server-row__dot--pinging" />
                : ping.online
                  ? <span className="server-row__dot server-row__dot--online" />
                  : <span className="server-row__dot server-row__dot--offline" />}
            </div>
            <div className="server-row__info">
              <span className="server-row__name">{server.name}</span>
              <span className="server-row__address">{server.host}:{server.port}</span>
              {ping?.online && ping.motd && (
                <span className="server-row__motd">{ping.motd}</span>
              )}
            </div>
            <div className="server-row__stats">
              {ping?.online
                ? <span>{ping.onlinePlayers}/{ping.maxPlayers} players · {ping.latencyMs}ms · {ping.version}</span>
                : ping ? <span className="text--muted">Offline</span>
                : <span className="text--muted">Pinging...</span>}
            </div>
            <button className="btn btn--ghost btn--sm" onClick={() => setEditingServer(server)}>Edit</button>
            <button className="btn btn--danger btn--sm" onClick={() => handleDelete(server)}>Delete</button>
          </div>
        );
      })}
    </div>
  )}

  {showAddModal && (
    <AddServerModal onClose={() => setShowAddModal(false)} onAdd={handleAdd} />
  )}
  {editingServer && (
    <EditServerModal
      server={editingServer}
      onClose={() => setEditingServer(null)}
      onUpdate={handleUpdate}
    />
  )}
</div>
```
