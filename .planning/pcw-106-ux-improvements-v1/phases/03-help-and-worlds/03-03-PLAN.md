```tsx
import React, { useEffect, useState, useRef } from 'react';
import { useNavigate } from 'react-router-dom';
import { useSelectedInstance } from '../../stores/selected-instance-context';

function timeAgo(ms: number): string {
  const seconds = Math.floor((Date.now() - ms) / 1000);
  if (seconds < 60) return 'just now';
  const minutes = Math.floor(seconds / 60);
  if (minutes < 60) return `${minutes}m ago`;
  const hours = Math.floor(minutes / 60);
  if (hours < 24) return `${hours}h ago`;
  return `${Math.floor(hours / 24)}d ago`;
}

interface WorldCardProps {
  world: WorldInfo;
  instanceId: string;
}

function WorldCard({ world, instanceId }: WorldCardProps) {
  const [expanded, setExpanded] = useState(false);
  const [notes, setNotes] = useState('');
  const [screenshots, setScreenshots] = useState<ScreenshotInfo[]>([]);
  const [lightboxSrc, setLightboxSrc] = useState<string | null>(null);
  const saveTimer = useRef<ReturnType<typeof setTimeout> | null>(null);

  useEffect(() => {
    if (!expanded) return;
    window.launcher.worlds.getNotes(instanceId, world.folderName).then(setNotes);
    window.launcher.worlds.listScreenshots(instanceId).then((all) =>
      // filter to screenshots that match this world (best-effort by date proximity — not possible without NBT)
      // show all screenshots for now
      setScreenshots(all)
    );
  }, [expanded, instanceId, world.folderName]);

  const handleNotesChange = (value: string) => {
    setNotes(value);
    if (saveTimer.current) clearTimeout(saveTimer.current);
    saveTimer.current = setTimeout(() => {
      window.launcher.worlds.setNotes(instanceId, world.folderName, value);
    }, 500);
  };

  return (
    <div className="world-card">
      <button
        className="world-card__header"
        onClick={() => setExpanded((e) => !e)}
      >
        <span className="world-card__name">{world.folderName}</span>
        <span className="world-card__meta">Last played {timeAgo(world.lastModified)}</span>
        <span className="world-card__chevron">{expanded ? '▲' : '▼'}</span>
      </button>

      {expanded && (
        <div className="world-card__body">
          <div className="world-card__notes-section">
            <span className="world-card__section-label">Notes</span>
            <textarea
              className="world-card__notes"
              value={notes}
              onChange={(e) => handleNotesChange(e.target.value)}
              placeholder="Add notes about this world..."
              rows={4}
            />
          </div>

          {screenshots.length > 0 && (
            <div className="world-card__screenshots-section">
              <span className="world-card__section-label">Screenshots ({screenshots.length})</span>
              <div className="world-card__screenshots">
                {screenshots.map((s) => (
                  <img
                    key={s.fileName}
                    className="world-card__thumb"
                    src={`launcher-file://${s.filePath}`}
                    alt={s.fileName}
                    onClick={() => setLightboxSrc(`launcher-file://${s.filePath}`)}
                  />
                ))}
              </div>
            </div>
          )}
        </div>
      )}

      {lightboxSrc && (
        <div className="lightbox" onClick={() => setLightboxSrc(null)}>
          <img className="lightbox__img" src={lightboxSrc} alt="Screenshot" />
        </div>
      )}
    </div>
  );
}

export function WorldsPage() {
  const { selectedInstanceId } = useSelectedInstance();
  const navigate = useNavigate();
  const [instances, setInstances] = useState<InstanceInfo[]>([]);
  const [worlds, setWorlds] = useState<WorldInfo[]>([]);
  const [loading, setLoading] = useState(false);

  useEffect(() => {
    window.launcher.instances.list().then(setInstances);
  }, []);

  useEffect(() => {
    if (!selectedInstanceId) { setWorlds([]); return; }
    setLoading(true);
    window.launcher.worlds.list(selectedInstanceId)
      .then(setWorlds)
      .finally(() => setLoading(false));
  }, [selectedInstanceId]);

  const selectedInstance = instances.find((i) => i.id === selectedInstanceId) ?? null;

  return (
    <div className="worlds">
      <div className="worlds__header">
        <h2>My Worlds</h2>
      </div>

      <div className="mods__instance-header">
        {selectedInstance ? (
          <>
            <span className="mods__instance-label">
              Instance: <strong>{selectedInstance.name}</strong> ({selectedInstance.versionId})
            </span>
            <button className="btn btn--ghost btn--sm" onClick={() => navigate('/instances')}>
              ← Change
            </button>
          </>
        ) : (
          <>
            <span className="mods__instance-label mods__instance-label--none">No instance selected</span>
            <button className="btn btn--ghost btn--sm" onClick={() => navigate('/instances')}>
              ← Select Instance
            </button>
          </>
        )}
      </div>

      {!selectedInstanceId && (
        <div className="worlds__empty"><p>Select an instance to view its worlds.</p></div>
      )}

      {selectedInstanceId && loading && (
        <div className="worlds__empty"><p>Loading worlds…</p></div>
      )}

      {selectedInstanceId && !loading && worlds.length === 0 && (
        <div className="worlds__empty">
          <p>No worlds found.</p>
          <p className="form-group__hint" style={{ marginTop: 8 }}>
            Launch the game and create a world — it will appear here.
          </p>
        </div>
      )}

      {worlds.map((world) => (
        <WorldCard key={world.folderName} world={world} instanceId={selectedInstanceId!} />
      ))}
    </div>
  );
}
```
