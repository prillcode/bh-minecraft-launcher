  <files>src/renderer/stores/selected-instance-context.tsx</files>
  <action>
    Create `src/renderer/stores/selected-instance-context.tsx`:

    ```tsx
    import React, { createContext, useContext, useState } from 'react';

    interface SelectedInstanceContextValue {
      selectedInstanceId: string;
      setSelectedInstanceId: (id: string) => void;
    }

    const SelectedInstanceContext = createContext<SelectedInstanceContextValue>({
      selectedInstanceId: '',
      setSelectedInstanceId: () => {},
    });

    export function SelectedInstanceProvider({ children }: { children: React.ReactNode }) {
      const [selectedInstanceId, setSelectedInstanceId] = useState('');
      return (
        <SelectedInstanceContext.Provider value={{ selectedInstanceId, setSelectedInstanceId }}>
          {children}
        </SelectedInstanceContext.Provider>
      );
    }

    export function useSelectedInstance() {
      return useContext(SelectedInstanceContext);
    }
    ```

    In `src/renderer/App.tsx`, import `SelectedInstanceProvider` and wrap the
    authenticated app shell (the `<HashRouter>` block) with it:

    ```tsx
    import { SelectedInstanceProvider } from './stores/selected-instance-context';

    // in the return for authenticated users:
    return (
      <SelectedInstanceProvider>
        <HashRouter>
          ...existing content...
        </HashRouter>
      </SelectedInstanceProvider>
    );
    ```

    **Auto-select on mount:** After loading instances, if `selectedInstanceId` is empty
    or not found in the loaded list, select the first instance:

    ```tsx
    useEffect(() => {
      window.launcher.instances.list().then(async (list) => {
        let loaded = list;
        if (list.length === 0) {
          try {
            const instance = await window.launcher.instances.createBlockhaven();
            loaded = [instance];
          } catch {
            loaded = [];
          }
        }
        setInstances(loaded);
        // Auto-select first if nothing is currently selected or selection no longer exists
        if (loaded.length > 0) {
          const stillExists = loaded.some((i) => i.id === selectedInstanceId);
          if (!stillExists) {
            setSelectedInstanceId(loaded[0].id);
          }
        }
      });
    }, []);
    ```

    **Card click = select:** Change the card `onClick` from opening the edit modal to
    selecting the instance. Add a dedicated ✏ edit button to the card actions:

    ```tsx
    <div
      key={inst.id}
      className={`instance-card${selectedInstanceId === inst.id ? ' instance-card--selected' : ''}`}
      onClick={() => !launching && setSelectedInstanceId(inst.id)}
      style={{ cursor: launching ? 'default' : 'pointer' }}
    >
      ...existing card content...
      <button
        className="instance-card__edit"
        onClick={(e) => { e.stopPropagation(); setEditingInstance(inst); }}
        disabled={launching !== null}
        aria-label={`Edit ${inst.name}`}
      >
        ✏
      </button>
      <button
        className="instance-card__play"
        onClick={(e) => { e.stopPropagation(); handleLaunch(inst.id); }}
        disabled={launching !== null}
      >
        {launching === inst.id ? '⏳' : '▶'}
      </button>
      <button
        className="instance-card__delete"
        onClick={(e) => { e.stopPropagation(); handleDelete(inst.id, inst.name); }}
        disabled={launching !== null}
        aria-label={`Delete ${inst.name}`}
      >
        ✕
      </button>
    </div>
    ```

    **Delete clears selection:** In `handleDelete`, after removing the instance from state,
    if the deleted instance was selected, select the first remaining instance (or clear):

    ```tsx
    const handleDelete = async (id: string, name: string) => {
      if (!confirm(`Delete "${name}"? This cannot be undone.`)) return;
      await window.launcher.instances.delete(id);
      setInstances((prev) => {
        const remaining = prev.filter((i) => i.id !== id);
        if (selectedInstanceId === id) {
          setSelectedInstanceId(remaining.length > 0 ? remaining[0].id : '');
        }
        return remaining;
      });
    };
    ```

    **handleLaunch also selects:** When launching from the card's ▶ button, ensure that
    instance becomes the selected one:

    ```tsx
    const handleLaunch = async (instanceId: string) => {
      setSelectedInstanceId(instanceId);
      setLaunching(instanceId);
      ...
    };
    ```
